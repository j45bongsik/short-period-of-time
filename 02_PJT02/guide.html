@include min-height100 AOS/iOS 각각의 min-height 적용 
@mixin min-height100 {
    min-height: $vh100;

    }
}
@include height100 AOS/iOS 각각의 height 적용 
@mixin height100 {
    height: $vh100;
    @supports (-webkit-touch-callout: none) {
        height: 100vh;
    }
}
